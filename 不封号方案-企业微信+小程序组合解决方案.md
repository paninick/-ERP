```java
// 2. 规则引擎提取叫车信息
@Service
public class CarOrderService {
    public CarOrder extractOrderFromMessage(String content) {
        CarOrder order = new CarOrder();
        
        // 提取时间
        Pattern timePattern = Pattern.compile("(\d{1,2}:\d{2}|明天|后天|今天)");
        Matcher timeMatcher = timePattern.matcher(content);
        if (timeMatcher.find()) {
            order.setTime(timeMatcher.group());
        }
        
        // 提取起点
        Pattern fromPattern = Pattern.compile("从(.+?)(到|去)");
        Matcher fromMatcher = fromPattern.matcher(content);
        if (fromMatcher.find()) {
            order.setFromAddr(fromMatcher.group(1));
        }
        
        // 提取终点
        Pattern toPattern = Pattern.compile("到(.+?)(的|，|。|$)");
        Matcher toMatcher = toPattern.matcher(content);
        if (toMatcher.find()) {
            order.setToAddr(toMatcher.group(1));
        }
        
        // 提取货物信息
        Pattern cargoPattern = Pattern.compile("(面料|辅料|成品|裁片|布料|纱线).*?(\d+)(公斤|kg|斤|箱|包|件)");
        Matcher cargoMatcher = cargoPattern.matcher(content);
        if (cargoMatcher.find()) {
            order.setCargoType(cargoMatcher.group(1));
            order.setCargoQty(cargoMatcher.group(2) + cargoMatcher.group(3));
        }
        
        // 提取联系人
        Pattern contactPattern = Pattern.compile("@(\S{2,4})|联系(\S{2,4})");
        Matcher contactMatcher = contactPattern.matcher(content);
        if (contactMatcher.find()) {
            order.setContact(contactMatcher.group(1) != null ? contactMatcher.group(1) : contactMatcher.group(2));
        }
        
        // 提取电话
        Pattern phonePattern = Pattern.compile("1[3-9]\\d{9}");
        Matcher phoneMatcher = phonePattern.matcher(content);
        if (phoneMatcher.find()) {
            order.setPhone(phoneMatcher.group());
        }
        
        return order;
    }
}
```

```javascript
// 1. 实时订单通知
const app = getApp()

Page({
  data: {
    orders: [],
    socketOpen: false
  },
  
  onLoad() {
    this.connectWebSocket()
  },
  
  connectWebSocket() {
    wx.connectSocket({
      url: 'wss://your-server.com/ws/driver',
      success: (res) => {
        console.log('WebSocket连接成功')
        this.setData({ socketOpen: true })
      }
    })
    
    wx.onSocketMessage((res) => {
      const message = JSON.parse(res.data)
      if (message.type === 'new_order') {
        this.addNewOrder(message.data)
        this.showOrderNotification(message.data)
      }
    })
  },
  
  addNewOrder(order) {
    const orders = [order, ...this.data.orders]
    this.setData({ orders })
  },
  
  showOrderNotification(order) {
    wx.showModal({
      title: '新订单',
      content: `从${order.fromAddr}到${order.toAddr}\n时间：${order.time}`,
      confirmText: '接单',
      cancelText: '忽略',
      success: (res) => {
        if (res.confirm) {
          this.acceptOrder(order.id)
        }
      }
    })
  },
  
  acceptOrder(orderId) {
    wx.request({
      url: 'https://your-server.com/api/order/accept',
      method: 'POST',
      data: { orderId, driverId: app.globalData.driverId },
      success: (res) => {
        if (res.data.success) {
          wx.showToast({ title: '接单成功' })
        }
      }
    })
  }
})
```
